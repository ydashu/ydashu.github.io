    #!/bin/bash

# 配置
BACKUP_DIR="/backups/wordpress"
DB_USER="db_user"     # 替换
DB_PASSWORD="db_password"   # 替换
WEB_ROOT="/usr/local/lsws/virtual_host_root/wordpress"
BACKUP_USER="root"  # 替换为实际用户，例如 www-data 或 admin
BACKUP_GROUP="root" # 替换为实际组，例如 www-data 或 admin
DATE=$(date +%F)
RETENTION_DAYS=30  # 保留天数
CLEAN_LOG="/var/log/wp_clean_backup.log"

# 确保文件存在
touch "$CLEAN_LOG"

# 网站配置数组（站点名称、数据库名称、文件目录相对路径）
# 格式：( "site_name:db_name:web_subpath" )
SITES=(
    "site1:wordpress_site1:site1"
    "site2:wordpress_site2:site2"
)

# 确保备份目录存在
for SITE_CONFIG in "${SITES[@]}"; do
    # 解析配置
    SITE_NAME=$(echo "$SITE_CONFIG" | cut -d':' -f1)
    mkdir -p "$BACKUP_DIR/$SITE_NAME/db/$DATE" "$BACKUP_DIR/$SITE_NAME/files/$DATE"
done

# 备份数据库和文件
for SITE_CONFIG in "${SITES[@]}"; do
    # 解析配置
    SITE_NAME=$(echo "$SITE_CONFIG" | cut -d':' -f1)
    DB_NAME=$(echo "$SITE_CONFIG" | cut -d':' -f2)
    WEB_SUBPATH=$(echo "$SITE_CONFIG" | cut -d':' -f3)

    # 数据库备份
    mysqldump -u "$DB_USER" -p"$DB_PASSWORD" "$DB_NAME" > "$BACKUP_DIR/$SITE_NAME/db/$DATE/$DB_NAME.sql"
    gzip "$BACKUP_DIR/$SITE_NAME/db/$DATE/$DB_NAME.sql"

    # 文件备份
    tar -zcvf "$BACKUP_DIR/$SITE_NAME/files/$DATE/${SITE_NAME}_files.tar.gz" -C "$WEB_ROOT/$WEB_SUBPATH" .
done

# 设置权限
chown -R "$BACKUP_USER:$BACKUP_GROUP" "$BACKUP_DIR"
chmod -R 700 "$BACKUP_DIR"

# 清理函数
clean_old_backups() {
    echo "===== 开始清理 $(date +'%F %T') =====" >> "$CLEAN_LOG"

    # 清理数据库备份
    find "$BACKUP_DIR" -type d -path "*/db/20??-??-??" -mtime +$RETENTION_DAYS \
        -exec echo "删除数据库备份: {}" >> "$CLEAN_LOG" \; \
        -exec rm -rf {} \;

    # 清理文件备份
    find "$BACKUP_DIR" -type d -path "*/files/20??-??-??" -mtime +$RETENTION_DAYS \
        -exec echo "删除文件备份: {}" >> "$CLEAN_LOG" \; \
        -exec rm -rf {} \;

    echo "===== 清理完成 =====" >> "$CLEAN_LOG"
}
# 执行清理
clean_old_backups

    #!/bin/bash

# 配置
BACKUP_DIR="/backups/wordpress"
WEB_ROOT="/usr/local/lsws/virtual_host_root/wordpress"
DB_USER="db_user"
DB_PASSWORD="db_password"
RESTORE_USER="nobody"  # 替换为实际用户，例如 www-data
RESTORE_GROUP="nobody" # 替换为实际组，例如 www-data
RESTORE_DATE="2025-04-18" # 替换为要恢复的日期，例如 2025-04-18
LOG_FILE="/var/log/wp_restore.log"

# 网站配置数组（站点名称、数据库名称、文件目录相对路径）
SITES=(
    "site1:wordpress_site1:site1"
    "site2:wordpress_site2:site2"
)

# 确保文件存在
touch "$LOG_FILE"

# 恢复数据库和文件
for SITE_CONFIG in "${SITES[@]}"; do
    # 解析配置
    SITE_NAME=$(echo "$SITE_CONFIG" | cut -d':' -f1)
    DB_NAME=$(echo "$SITE_CONFIG" | cut -d':' -f2)
    WEB_SUBPATH=$(echo "$SITE_CONFIG" | cut -d':' -f3)

    # 检查备份文件是否存在
    DB_BACKUP="$BACKUP_DIR/$SITE_NAME/db/$RESTORE_DATE/$DB_NAME.sql.gz"
    FILE_BACKUP="$BACKUP_DIR/$SITE_NAME/files/$RESTORE_DATE/${SITE_NAME}_files.tar.gz"

    if [ ! -f "$DB_BACKUP" ]; then
        echo "Error: Database backup $DB_BACKUP not found for $SITE_NAME" >> "$LOG_FILE"
        continue
    fi
    if [ ! -f "$FILE_BACKUP" ]; then
        echo "Error: File backup $FILE_BACKUP not found for $SITE_NAME" >> "$LOG_FILE"
        continue
    fi

    # 恢复数据库
    echo "Restoring database $DB_NAME at $(date)..." >> "$LOG_FILE"
    mysql -u "$DB_USER" -p"$DB_PASSWORD" -e "DROP DATABASE IF EXISTS $DB_NAME; CREATE DATABASE $DB_NAME;" >> "$LOG_FILE" 2>&1
    gunzip < "$DB_BACKUP" | mysql -u "$DB_USER" -p"$DB_PASSWORD" "$DB_NAME" >> "$LOG_FILE" 2>&1
    if [ $? -eq 0 ]; then
        echo "Database $DB_NAME restored successfully." >> "$LOG_FILE"
    else
        echo "Error: Failed to restore database $DB_NAME." >> "$LOG_FILE"
    fi

    # 备份现有文件（可选）
    if [ -d "$WEB_ROOT/$WEB_SUBPATH" ]; then
        mv "$WEB_ROOT/$WEB_SUBPATH" "$WEB_ROOT/${WEB_SUBPATH}_backup_$(date +%F)" >> "$LOG_FILE" 2>&1
    fi

	# 创建目标目录（如果不存在） 
	mkdir -p "$WEB_ROOT/$WEB_SUBPATH" >> "$LOG_FILE" 2>&1

    # 恢复文件
    echo "Restoring files for $SITE_NAME at $(date)..." >> "$LOG_FILE"
    tar -zxvf "$FILE_BACKUP" -C "$WEB_ROOT/$WEB_SUBPATH/" >> "$LOG_FILE" 2>&1
    if [ $? -eq 0 ]; then
        echo "Files for $SITE_NAME restored successfully." >> "$LOG_FILE"
    else
        echo "Error: Failed to restore files for $SITE_NAME." >> "$LOG_FILE"
    fi

    # 设置权限（仅对存在的目录） 不需要
    #if [ -d "$WEB_ROOT/$WEB_SUBPATH" ]; then
        #chown -R "$RESTORE_USER:$RESTORE_GROUP" "$WEB_ROOT/$WEB_SUBPATH" >> "$LOG_FILE" 2>&1
        #chmod -R 755 "$WEB_ROOT/$WEB_SUBPATH" >> "$LOG_FILE" 2>&1
        #echo "Permissions set for $WEB_ROOT/$WEB_SUBPATH." >> "$LOG_FILE"
    #else
        #echo "Warning: Directory $WEB_ROOT/$WEB_SUBPATH does not exist, skipping permissions." >> "$LOG_FILE"
    #fi
done

# 重启 LiteSpeed
echo "Restarting LiteSpeed at $(date)..." >> "$LOG_FILE"
/usr/local/lsws/bin/lswsctrl restart >> "$LOG_FILE" 2>&1

echo "Restore completed at $(date). Check $LOG_FILE for details." >> "$LOG_FILE"
echo "Restore completed. Check $LOG_FILE for details."
